### Basic Workflow

```python
import crystlib
import orilib
import plotlib
import numpy as np

# 1. Create lattice
lattice = crystlib.cubic_lattice_vec(a=3.6)

# 2. Create orientation
angles = np.radians([45, 30, 60])
R = orilib.np_euler_matrix(*angles)

# 3. Visualize
p = plotlib.plotter()
p.plotProj(ProjType='equalarea', sphere='half')
p.setAttributes(dirs=[[1,0,0], [1,1,0], [1,1,1]])
p.plotDirsNorms()
p.figsave('pole_figure.png')
```
